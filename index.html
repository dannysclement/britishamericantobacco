const express = require('express'); const multer = require('multer'); const cors = require('cors'); const path = require('path'); const app = express(); const PORT = 3000; // Middleware app.use(cors()); app.use(express.static('uploads')); // Set up storage for multer const storage = multer.diskStorage({ destination: (req, file, cb) => { cb(null, 'uploads/'); }, filename: (req, file, cb) => { cb(null, Date.now() + path.extname(file.originalname)); // Append extension } }); const upload = multer({ storage: storage }); // Upload endpoint app.post('/upload', upload.single('file'), (req, res) => { if (req.file) { res.json({ url: `http://localhost:${PORT}/${req.file.filename}` }); } else { res.status(400).json({ error: 'File upload failed' }); } }); // Start the server app.listen(PORT, () => { console.log(`Server is running on http://localhost:${PORT}`); });
